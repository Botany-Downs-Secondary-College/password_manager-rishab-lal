def askage():
    age = -1

    while type(age) != int or 0 >= age or age > 111:
        try:
            age = int(input("please enter a valid age: "))
        except:
            print("age invalid")

        if type(age) == int and 0 < age and age <= 111:
            age = int(age)
            return age
